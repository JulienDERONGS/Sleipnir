#------------------------------------------------------------
# Table: User
#------------------------------------------------------------

CREATE TABLE User(
        user_email              Varchar NOT NULL,
        user_password           Varchar (128) NOT NULL,
        userLevel_id            Int NOT NULL,
        PRIMARY KEY (user_email)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: Equipment
#------------------------------------------------------------

CREATE TABLE Equipment(
        equip_id                int (11) Auto_increment  NOT NULL,
        equip_name              Varchar (255) NOT NULL,
        equipType_id            Int NOT NULL,
        PRIMARY KEY (equip_id)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: UserLevel
#------------------------------------------------------------

CREATE TABLE UserLevel(
        userLevel_id            int (11) Auto_increment  NOT NULL,
        userLevel_name          Varchar (128) NOT NULL,
        PRIMARY KEY (userLevel_id)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: EquipmentType
#------------------------------------------------------------

CREATE TABLE EquipmentType(
        equipType_id            int (11) Auto_increment  NOT NULL,
        equipType_name          Varchar NOT NULL,
        PRIMARY KEY (equipType_id)
)ENGINE=InnoDB;

ALTER TABLE User ADD CONSTRAINT FK_User_userLevel_id FOREIGN KEY (userLevel_id) REFERENCES UserLevel(userLevel_id);
ALTER TABLE Equipment ADD CONSTRAINT FK_Equipment_equipType_id FOREIGN KEY (equipType_id) REFERENCES EquipmentType(equipType_id);
